### 12. this指向的具体规则？

```javascript
var name = '寒雨', age = 20;
function hello() {
    window.console.log(this.name, ':', this.age);
}
hello();// hy 20

hello.bind({name: '2', age: '30'})();// 2 30

let obj = {
    sayHello: function () {
        window.console.log(this);
    },
    sayHi: ()=> {
        window.console.log(this);
    }
};

obj.sayHello(); // 调用主体obj
obj.sayHi(); // window 定义时的代码块
```
